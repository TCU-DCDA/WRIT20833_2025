Complete Analysis Workflow

import pandas as pd import matplotlib.pyplot as plt from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer def full_analysis(csv_file, text_column): # 1. Load data df = pd.read_csv(csv_file) # 2. Clean data df = df.dropna(subset=[text_column]) # 3. Analyze sentiment analyzer = SentimentIntensityAnalyzer() df['sentiment'] = df[text_column].apply( lambda x: analyzer.polarity_scores(x)['compound'] ) # 4. Visualize results plt.figure(figsize=(12, 8)) plt.hist(df['sentiment'], bins=30, alpha=0.7) plt.title('Sentiment Distribution') plt.xlabel('Sentiment Score') plt.ylabel('Frequency') plt.show() # 5. Summary statistics return df['sentiment'].describe()
